#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define MAXSIZE 100
typedef struct
{
	int key;
}record;
typedef struct
{
	record r[MAXSIZE+1];
	int length;
}Sqlist;

void Initsqlist(Sqlist &L)/*初始化线性表*/ 
{
	int i; 
	printf("请输入你要排序的数字的个数:\n");
	scanf("%d",&L.length);
	printf("请输入你的数据:\n");
	for(i=1;i<=L.length;i++) 
	scanf("%d",&L.r[i].key);
		
} 

void InsertSort(Sqlist &L)/*书上的直接插入排序*/ 
{
	int i,j;
	for(i=2;i<=L.length;i++) 
	{
		if(L.r[i].key<L.r[i-1].key)/*递增为有序*/ 
		{
			L.r[0]=L.r[i];/*复制哨兵*/ 
			L.r[i]=L.r[i-1];
		for(j=i-2;L.r[0].key<L.r[j].key;--j)
			L.r[j+1]=L.r[j];/*记录后移*/ 
		L.r[j+1]=L.r[0];/*插入到正确位置*/ 
		}
	}
} 

void Insertsort(Sqlist &L)/*老师ppt的代码,有错误害我改了半小时*/

	{

			for(j=i-1;L.r[0].key<L.r[j].key;--j)
				L.r[j+1]=L.r[j];/*记录后移*/
			L.r[j+1]=L.r[0];/*插入到正确位置*/
		}
	}	
} 

void Binsertsort(Sqlist &L)/*折半插入排序*/ 
{
	int i,j,low,high,m;
	for(i=2;i<=L.length;i++)
	{
		L.r[0]=L.r[i];/*复制哨兵*/ 
		low=1;	high=i-1;
		while(low<=high)/*寻找插入位置*/ 
		{
			m=(high+low)/2;//折半
			if(L.r[0].key<L.r[m].key)
			high=m-1;
			else
			low=m+1;
		}
		for(j=i-1;j>=high+1;--j)/*记录后移*/ 
		{
			L.r[j+1]=L.r[j];
		}
		L.r[high+1]=L.r[0];/*插入*/ 
	}
}

void print(Sqlist &L)/*输出线性表*/ 
{
	int i;
	for(i=1;i<=L.length;i++)
	{
		printf("%4d",L.r[i].key); 
	} 
	printf("\n");
}

int partition(Sqlist &L,int low,int high)/*记录枢轴的位置*/ 
{
	int ptkey; 
	L.r[0]=L.r[low];/*用子表的第一个记录做枢轴记录*/
	ptkey=L.r[low].key;/*枢轴记录关键字*/
	while(low<high)
	{
		while(low<high&&L.r[high].key>=ptkey)/*将比枢轴记录小的记录移到low*/ 
			--high;
		L.r[low]=L.r[high];
		while(low<high&&L.r[low].key<=ptkey)/*将比枢轴记录大的记录移到low*/
			++low;
		L.r[high]=L.r[low]; 
	}
	L.r[low]=L.r[0];/*记录枢轴位置*/ 
	return low;/*返回枢轴位置*/ 
	
}

void Qsort(Sqlist &L,int low,int high)/*递归快速排序核心算法*/ 
{
	int prt;
	if(low<high)
	{
		prt=partition(L,low,high);/*将线性表一分为二*/ 
		Qsort(L,low,prt-1);/*将左边排序*/ 
		Qsort(L,prt+1,high);/*将右边排序*/	
	}	
} 

void Quicksort(Sqlist &L)/*快速排序*/ 
{
	Qsort(L,1,L.length);	
} 
void shellInsert(Sqlist &L,int dk)/*希尔排序算法*/ 
{
	//前后记录位置的增量是dk,而不是1
	//L.r[0]只是暂存单元,不是哨兵,当j<=0时插入位置已找到*/
	int i,j;
	for(i=dk+1;i<=L.length;i++)
		if(L.r[i].key<L.r[i-dk].key)
		{
			L.r[0]=L.r[i];/*暂存在l.r[0]*/ 
		for(j=i-dk;j>0&&(L.r[0].key<L.r[j].key);j-=dk) 
				L.r[j+dk]=L.r[j];/*记录后移,查找插入位置*/ 
	 	L.r[j+dk]=L.r[0];
	 	}
} 

void ShellSort(Sqlist &L,int t)
{
	int k;
	int data[3]={5,3,1}; 
	for(k=0;k<t;++k)
	{
		shellInsert(L,data[k]);	
	}	
} 

void swap(int *a,int *b)/*交换算法*/ 
{ 
	int temp;
	temp=*a;
	*a=*b;
	*b=temp; 
}

void selectsort(Sqlist &L)
{
	int i,j,min;
	for(i=1;i<=L.length;i++)
	{
		min=i;/*假设当前值是最小的*/
		for(j=i+1;j<=L.length;j++){
			if(L.r[j].key<L.r[min].key)
				min=j;/*找到后面序列中最小的数*/ 
			}
		if(min!=i)
		swap(&L.r[i].key,&L.r[min].key);  
	}	
}

}
