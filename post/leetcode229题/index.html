题目

给定一个大小为 n 的数组，找出其中所有出现超过 ⌊ n/3 ⌋ 次的元素。

说明: 要求算法的时间复杂度为 O(n)，空间复杂度为 O(1)。

示例 1:

输入: [3,2,3]
输出: [3]

示例 2:

输入: [1,1,1,3,3,2,2,2]
输出: [1,2]

题解

1.投票法

由于要求找到的是出现次数超过⌊ n/3 ⌋ 的元素,所以最多可以找到两个不同的数字,假定两个结果候选人者,遍历数组中的所有元素.

  • 如果是候选者,则对应票数加1
  • 都不是且当前候选者的计数不为0,减少一票
  • 都不是且当前候选者的计数为0,设置当前数为新的候选者

最后统计两个候选者的票数是否超过⌊ n/3 ⌋

C语言实现(建议不要用c语言刷题,太麻烦)

int *majorityElement(int *nums, int numsSize, int *returnSize)
{
    int x = 0, y = 0, cx = 0, cy = 0,i=0;
    if (numsSize == 0)
    {
        *returnSize = numsSize;
        return nums;
    }
    for (i = 0; i < numsSize; i++)
    {
        if (nums[i] == x)
        {
            ++cx;
        }
        else if (nums[i] == y)
        {
            ++cy;
        }
        else if (cx == 0)
        {
            x = nums[i];
            ++cx;
        }
        else if (cy == 0)
        {
            y = nums[i];
            ++cy;
        }
        else
        {
            --cx;
            --cy;
        }
    }
    cx = cy = 0;
    for (i = 0; i < numsSize; i++)
    {
        if (nums[i] == x)
            ++cx;
        else if (nums[i] == y)
            ++cy;
        else
            continue;
    }
    *returnSize = 0;
    int *res = malloc(sizeof(int) * 2);
    if (cx > numsSize / 3)
    {
        res[(*returnSize)++] = x;

    }
    if (cy > numsSize / 3)
    {
        res[(*returnSize)++] = y;
    }
    return res;
}

vector<int> majorityElement(vector<int> &nums)
{
    int size = nums.size();
    //候选者x,y;票数cx,cy
    int x = 0, y = 0, cx = 0, cy = 0;
    vector<int> res;
    if (nums.size() == 0)
        return res;
    for (int i = 0; i < nums.size(); i++)
    {
        if (nums[i] == x)
        {
            ++cx;
        }
        else if (nums[i] == y)
        {
            ++cy;
        }
        else if (cx == 0)
        {
            x = nums[i];
            ++cx;
        }
        else if (cy == 0)
        {
            y = nums[i];
            ++cy;
        }
        else
        {
            --cx;
            --cy;
        }
    }
    cx = cy = 0;
    //计算候选者的票数
    for (auto i : nums)
    {
        if (i == x)
            ++cx;
        else if (i == y)
            ++cy;
        else
            continue;
    }
    if (cx > nums.size() / 3)
    {
        res.push_back(x);
    }
    if (cy > nums.size() / 3)
    {
        res.push_back(y);
    }
    return res;
}

2.暴力破解

先将数组进行排序,计数,判断即可

c++实现

vector<int> majorityElement(vector<int>& nums) {
        int len = nums.size();
        vector<int> res;
        if(len == 0)
            return nums;
        sort(nums.begin(),nums.end());
        int count = 0;
        int cur = nums[0];
        for(int i = 0;i < len;i++)
        {
            if(nums[i] == cur)
            {
                count++;
            }
            else{
                if(count > (len / 3))
                {
                    res.push_back(cur);
                }
                count = 1;//注意这里计数为1
                cur = nums[i];
            }
        }
        if(count > (len / 3) ) res.push_back(cur);
        return res;
}

java实现

class Solution {
    public List<Integer> majorityElement(int[] nums) {
        List<Integer> li = new ArrayList<>();
        int len = nums.length;
        if (len == 0) {
            return li;
        }
        Arrays.sort(nums);
        int cur = nums[0];
        int count = 0;
        for (int i = 0; i < len; i++) {
            if(cur == nums[i]){
                count++;
            }
            else{
                if (count > (len / 3))
                {
                    li.add(cur);
                }
                count = 1;
                cur = nums[i];
            }
        }
        if (count > (len / 3)) li.add(cur);
        return li;
    }
}
