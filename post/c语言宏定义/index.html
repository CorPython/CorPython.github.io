c语言宏定义初识

带参数的宏的原则:

  • 一切都要括号
  • 整个值要括号
  • 参数出现的每个地方都要括号

注意:结尾千万不要加上分号

#include<stdio.h>
#include<stdlib.h>

#define MAX( x, y ) ( ((x) > (y)) ? (x) : (y) )
#define MIN( x, y ) ( ((x) < (y)) ? (x) : (y) )
#define UPCASE( c ) ( ((c) >= 'a' && (c) <= 'z') ? ((c) - 0x20) : (c) )//大小写转换
#define Cube(x) ((x) * (x) * (x) )
#define func(x) ((x) / 3)

#define PI 3.14
//#undef PI 撤销宏定义
int main()
{
    printf("%.2f\n",PI);
    printf("%d\n",MAX(15,20));
    printf("%.1f\n",MAX(15.5,20.5));
    printf("%c\n",UPCASE('a'));
    printf("%c\n",UPCASE('b'));
    printf("%c\n",UPCASE('c'));
    printf("%c\n",UPCASE('D'));
    printf("%c\n",UPCASE('E'));
    printf("%d\n",Cube(5));
    printf("%d\n",func(5 + 4));
    return 0;
}
