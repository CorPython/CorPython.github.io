3.冒泡排序

核心代码:

for (int i = 1; i < n; i++) {
      for (int j = 1; j < n - i - 1; j++) {
         if (a[j - 1] > a[j])
         { 
             temp = a[j - 1];
             a[j - 1] = a[j];
             a[j] = temp;
         }
      }
   }

时间复杂度： $O\left(n^{2}\right)$

空间复杂度： $O\left(n^{2}\right)$

稳定

4.选择排序

核心代码:

void selectSort(int a[], int len)
{
   int mindex, temp;
   for(int i = 0; i<len-1;i++)
   {
       mindex = i; //假设当前数字是最小值
       for(int j = i+1; j<len; j++)
      {
          if(a[j]<a[mindex])//找到后面的数字中最小的那个
            mindex = j;
      
      }
      temp = a[i];
      a[i] = a[mindex];
      a[mindex] = temp;
   }
}

空间复杂度：$O\left(1{}\right)$

最坏时间复杂度： $O\left(n^{2}\right)$

最好时间复杂度: $O\left(n^{2}\right)$

平均时间复杂度：$O\left(n^{2}\right)$

不稳定

5.快速排序

#include <iostream>

using namespace std;

void quickSort(int a[], int low , int high)
{
   if (low < high) //判断是否满足排序条件，递归的终止条件
   {
      int i = low, j = high;
      int x = a[low];
      //将待排序数组的第一个元素作为哨兵，将数组划分为大于哨兵以及小于哨兵的两部分
      while (i < j)
      {
         while (i < j && a[j] >= x) j--; //从最右侧元素开始，如果比哨兵大，那么它的位置就正确，然后判断前一个元素，直到不满足条件
         if (i < j) a[i++] = a[j]; //把不满足位次条件的那个元素值赋值给哨兵,并把i的加1
         while (i < j && a[i] <= x) i++; //换成左侧下标为i的元素开始与哨兵比较大小，比其小，那么它所处的位置就正确，然后判断后一个，直到不满足条件
         if (i < j) a[j--] = a[i]; //把不满足位次条件的那个元素值赋值给下标为j的元素，（下标为j的元素已经保存到前面，不会丢失）并把j的加1
      }
      a[i] = x;   //完成一次排序，把哨兵赋值到下标为i的位置，即前面的都比它小，后面的都比它大
      quickSort(a, low , i - 1); //递归进行哨兵前后两部分元素排序 ， low,high的值不发生变化，i处于中间
      quickSort(a, i + 1 , high);
   }
}
int main(int argc, char const *argv[])
{
   int a[10] = {9, 6, 3, 8, 5, 2, 7, 4, 1, 0};
   quickSort(a, 0, 9);
   for (int i = 0; i < 10; i++)
   {
      cout << a[i] << endl;
   }
   return 0;
}

vector

必须包含头文件< vector >

vector初始化

vector< T >v1 v1是一个空vector，它潜在的元素是T类型的，执行默认初始化
vector< T >v2（v1） v2中包含有v1所有元素的副本
vector< T > v2=v1 等价于v2(v1),v2中包含有v1所有元素的副本
vector< T >v3(n,val) v3包含了n个重复的元素，每个元素的值都是val
vector< T >v4(n) v4包含了n个重复地执行了值初始化的对象
vector< T >v5{a，b，c… } v5包含了初始值个数的元素,每个元素被赋予相应的初始值
vector< T >v5={a，b，c…} 等价于v5{a，b，c…} 
vector<string> v1;
vector<int> v2(v1);
vector<int> v3;
vector<int> v4 = v3;
vector<int> v5 = { 1,2,3,4,5 }; //列表初始化,注意使用的是花括号
vector<string> v6 = { "hi","my","name","is","lee" };
vector<int> v7(5, -1); //初始化为-1,-1,-1,-1,-1。第一个参数是数目，第二个参数是要初始化的值
vector<int> v8(10); //默认初始化为0
vector<int> v9(4); //默认初始化为空字符串

vector常见用法

v.empty() 如果v不含有任何元素,返回真;否则返回假
v.size() 返回v中元素的个数
v.push_back(t) 向v的尾端添加一个值为t的元素
v[n] 和数组访问方法一致
v1=v2 将v2中的元素拷贝给v1
v1==v2 当且仅当它们的元素数量相同且对应位置的元素值都相同
v.begin() 返回一个当前vector容器中起始元素的迭代器。
v.end() 指向的是最后一个元素的下一个位置
v.front() 返回当前vector容器中起始元素的引用
v.back() 返回当前vector容器中末尾元素的引用

vector使用迭代器等访问的四种方式

注意:但凡使用了迭代器的循环体，都不要向迭代器所属的容器添加元素！

#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> v;
    for (int i = 0; i < 10; ++i)
    {
        v.push_back(i);
    }

    for (vector<int>::iterator i = v.begin(); i != v.end(); ++i)
    {
        cout << *i << " ";
    }
    cout << endl;

    for (vector<int>::reverse_iterator i = v.rbegin(); i != v.rend(); ++i)
    {
        cout << *i << " ";
    }
    cout << endl;

    for (auto i = v.cbegin(); i < v.cend(); i++)
    {
        cout << *i << ' ';
    }
    cout << endl;

    for (auto i : v)
    {
        cout << i << ' ';
    }
    cout << endl;
    return 0;
}

输出结果:

1 2 3 4 5 6
6 5 4 3 2 1

vector常用增删操作

函数 说明
push_back(i) 在尾部添加元素
pop_back() 删除尾部元素
clear() 删除所有元素
insert() 插入到指定位置 
#include <iostream>
#include <vector>
#include <string>

using namespace std;

template <typename T>
void showvector(vector<T> v)
{
    cout << "当前元素有: ";
    for (typename std::vector<T>::iterator i = v.begin(); i != v.end(); i++)
    {
        cout << *i << " ";
    }
    cout << endl;
}

int main()
{
    vector<string> v6 = { "hi", "my", "name", "is", "lee"};
    v6.resize(3);  //重新调整vector容量大小
    showvector(v6);

    vector<int> v5 = { 1, 2, 3, 4, 5 }; //列表初始化,注意使用的是花括号
    //访问第一个元素
    cout << v5.front() << endl;
    cout << *(v5.begin()) << endl;
    //访问最后一个元素
    cout << v5.back() << endl;
    cout << *(v5.end() - 1) << endl;

    showvector(v5);
    v5.pop_back(); //删除最后一个元素
    showvector(v5);
    v5.push_back(6); //加入一个元素并把它放在最后
    showvector(v5);
    v5.insert(v5.begin() + 1, 9); //在第二个位置插入新元素
    showvector(v5);
    v5.erase(v5.begin() + 3);  //删除第四个元素
    showvector(v5);
    v5.insert(v5.begin() + 1, 7, 8); //连续插入7个8
    showvector(v5);
    v5.clear(); //清除所有内容
    showvector(v5);
    return 0;
}

warning: ISO C++ forbids converting a string constant to 'char*' [-Wwrite-strings]

1.char const *s = "hello";

for循环访问(和vector差不多)

四种方式

#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main(int argc, char const *argv[])
{
    string s = "hello,world!";
    for (size_t i = 0; i < s.length(); i++)
    {
        cout << s[i] ;
    }
    cout << endl;
    
    for (string::iterator i = s.begin(); i != s.end(); ++i)
    {
        cout << *i ;
    }
    cout << endl;
    
    for (auto i = s.begin(); i < s.end(); i++)
    {
        cout << *i ;
    }
    cout << endl;
    
    for (auto i : s)
    {
        cout << i ;
    }
    cout << endl;
    return 0;
}

注:使用下列代码可能会出现警告

for (int i = 0; i < s.length(); ++i)
{
        cout << s[i] ;
}

warning: comparison of integer expressions of different signedness: 'int' and 'std::__cxx11::basic_string<char>::size_type' {aka 'long long unsigned int'} [-Wsign-compare]

因为:string::length() 返回值为size_t类型为unsigned int,警告是因为int 类型跟unsigned int类型比较，类型不对。只要将int修改为size_t即可

stack

基础操作

push 入栈一个元素
pop 出栈一个元素，pop无返回值
top 取栈顶元素
size 查看元素个数
to_string(num) 将数字转化为字符串(c++11) 
#include <iostream>
#include <algorithm>
#include <stack>
using namespace std;

int main(int argc, char const *argv[])
{
    stack<int> s;

    for (int i = 0; i < 5; ++i)
    {
        s.push(i + 1);
    }
    //访问栈内所有元素
    for (size_t i = 0; i < s.size(); i++)
    {
        cout << s.top() << endl;
        s.pop();
    }

    while (!s.empty())
    {
        cout << s.top() << endl;
        s.pop();
    }
    return 0;
}

//十进制转换为二进制
int itob(int decimal) {
    stack<int> s;
    int res = 0;
    while (decimal != 0) {
        s.push(decimal % 2);
        decimal /= 2;
    }
    while (!s.empty()) {
        res = res * 10 + s.top();
        s.pop();
    }
    return res;
}
