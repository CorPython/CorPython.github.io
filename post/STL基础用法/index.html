STL:Standard Template Library 标准模板库

序列式容器[Sequential Containers]

  • Array(数组)
  • Vector(单向变长数组)
  • Deque(双向变长数组)
  • List(双向链表)
  • Forward-List(单向链表)

关联式容器[Associative Containers]

  • Set/Multiset(不重复/可重复集合)
  • Map/Multimap(不重复/可重复哈希表)

无序容器[Unordered Containers]

  • unordered Set/Multiset
  • unordered Map/Multimap

容器适配器[Container Adapter]

  • stack
  • queue
  • priority_queue

数据结构对应头文件

向量[vector] 连续存储的元素

列表[list] 由节点组成的双向链表，每个结点包含着一个元素

双队列[deque] 连续存储的指向不同元素的指针所组成的数组

集合[set] 由节点组成的红黑树，每个节点都包含着一个元素，节点之间以某种作用于元素对的谓词排列，没有两个不同的元素能够拥有相同的次序

多重集合[multiset] 允许存在两个次序相等的元素的集合

栈[stack] 后进先出的值的排列

队列[queue] 先进先出的执的排列

优先队列[priority_queue] 元素的次序是由作用于所存储的值对上的某种谓词决定的的一种队列

映射[map] 由{键，值}对组成的集合，以某种作用于键对上的谓词排列

vector初始化

vector< T >v1 v1是一个空vector，它潜在的元素是T类型的，执行默认初始化
vector< T >v2（v1） v2中包含有v1所有元素的副本
vector< T > v2=v1 等价于v2(v1),v2中包含有v1所有元素的副本
vector< T >v3(n,val) v3包含了n个重复的元素，每个元素的值都是val
vector< T >v4(n) v4包含了n个重复地执行了值初始化的对象
vector< T >v5{a，b，c… } v5包含了初始值个数的元素,每个元素被赋予相应的初始值

vector常见用法

v.empty() 如果v不含有任何元素,返回真;否则返回假
v.size() 返回v中元素的个数
v.push_back(t) 向v的尾端添加一个值为t的元素
v[n] 和数组访问方法一致
v1=v2 将v2中的元素拷贝给v1
v1==v2 当且仅当它们的元素数量相同且对应位置的元素值都相同
v.begin() 返回一个当前vector容器中起始元素的迭代器。
v.end() 指向的是最后一个元素的下一个位置
v.front() 返回当前vector容器中起始元素的引用
v.back() 返回当前vector容器中末尾元素的引用

vector使用迭代器等访问的四种方式

注意:但凡使用了迭代器的循环体，都不要向迭代器所属的容器添加元素！

#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> v;
    for (int i = 0; i < 10; ++i)
    {
        v.push_back(i);
    }

    for (vector<int>::iterator i = v.begin(); i != v.end(); ++i)
    {
        cout << *i << " ";
    }
    cout << endl;

    for (vector<int>::reverse_iterator i = v.rbegin(); i != v.rend(); ++i)
    {
        cout << *i << " ";
    }
    cout << endl;

    for (auto i = v.cbegin(); i < v.cend(); i++)
    {
        cout << *i << ' ';
    }
    cout << endl;

    for (auto i : v)
    {
        cout << i << ' ';
    }
    cout << endl;
    return 0;
}

输出结果:

1 2 3 4 5 6
6 5 4 3 2 1

{
    cout << "当前元素有: ";

    {
        cout << *i << " ";
    }

}

int main()
{
    vector<string> v6 = { "hi", "my", "name", "is", "lee"};
    v6.resize(3);  //重新调整vector容量大小
    showvector(v6);

    vector<int> v5 = { 1, 2, 3, 4, 5 }; //列表初始化,注意使用的是花括号
    //访问第一个元素
    cout << v5.front() << endl;
    cout << *(v5.begin()) << endl;
    //访问最后一个元素
    cout << v5.back() << endl;
    cout << *(v5.end() - 1) << endl;

    showvector(v5);
    v5.pop_back(); //删除最后一个元素
    showvector(v5);
    v5.push_back(6); //加入一个元素并把它放在最后
    showvector(v5);
    v5.insert(v5.begin() + 1, 9); //在第二个位置插入新元素
    showvector(v5);
    v5.erase(v5.begin() + 3);  //删除第四个元素
    showvector(v5);
    v5.insert(v5.begin() + 1, 7, 8); //连续插入7个8
    showvector(v5);
    v5.clear(); //清除所有内容
    showvector(v5);
    return 0;
}

迭代器

迭代器 说明
[begin]/[cbegin] 返回指向容器第一个元素的迭代器
[end]/[cend] 返回指向容器尾端的const迭代器
[rbegin]/[crbegin] 返回指向容器最后元素的逆向迭代器
[rend]/[crend] 返回指向前端的逆向const迭代器 
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> v={1,2,3,4,5,6,7,8};

    for (vector<int>::iterator i = v.begin(); i != v.end(); ++i)
    {
        cout << (*i) << endl;
    }
    
    for (vector<int>::const_iterator i = v.cbegin(); i != v.cend(); ++i)
    {
        cout << (*i) << endl;
    }
   
    for (vector<int>::const_reverse_iterator i = v.crbegin(); i != v.crend(); ++i)
    {
        cout << (*i) << endl;
    }
  
    for (vector<int>::reverse_iterator i = v.rbegin(); i != v.rend(); ++i)
    {
        cout << (*i) << endl;
    }

    return 0;
}

stack

基础操作

push 入栈一个元素
pop 出栈一个元素，pop无返回值
top 取栈顶元素
size 查看元素个数
to_string(num) 将数字转化为字符串(c++11) 
#include <iostream>
#include <algorithm>
#include <stack>
using namespace std;

int main(int argc, char const *argv[])
{
    stack<int> s;

    for (int i = 0; i < 5; ++i)
    {
        s.push(i + 1);
    }
    //访问栈内所有元素
    for (size_t i = 0; i < s.size(); i++)
    {
        cout << s.top() << endl;
        s.pop();
    }

    while (!s.empty())
    {
        cout << s.top() << endl;
        s.pop();
    }
    return 0;
}

//十进制转换为二进制
int itob(int decimal) {
    stack<int> s;
    int res = 0;
    while (decimal != 0) {
        s.push(decimal % 2);
        decimal /= 2;
    }
    while (!s.empty()) {
        res = res * 10 + s.top();
        s.pop();
    }
    return res;
}

由于multimap可以包含重复的key,所以对于删除和查找，erase（key）会删除掉所有key的map，查找find（key）返回第一个key的迭代器

    //输出B的value

插入元素

deque块在头部和尾部都可已插入和删除元素，而不需要移动其它元素。使用push_back()方法在尾部插入元素，使用push_front()方法在首部插入元素，使用insert()方法在中间插入元素

#include <iostream>
#include <deque>
#include <string>

using namespace std;

int main(int argc, char const *argv[])
{
    deque<int> d = {1, 2, 3, 4, 5, 6};


    for (deque<int>::iterator i = d.begin(); i != d.end(); ++i)
    {
        cout << (*i) << endl;
    }

    d.push_front(0);
    d.push_back(7);
    cout << "============" << endl;

    //正向遍历
    for (deque<int>::iterator i = d.begin(); i != d.end(); ++i)
    {
        cout << (*i) << endl;
    }
    cout << "============" << endl;

    //反向遍历
    for (deque<int>::reverse_iterator i = d.rbegin(); i != d.rend(); ++i)
    {
        cout << (*i) << endl;
    }
    cout << "============" << endl;
    
    //在任意位置插入
    d.insert(d.begin() + 3, 999);
    for (deque<int>::iterator i = d.begin(); i != d.end(); ++i)
    {
        cout << (*i) << endl;
    }
    cout << "============" << endl;
    return 0;
}

删除元素

#include <iostream>
#include <deque>
#include <string>

using namespace std;

int main(int argc, char const *argv[])
{
    deque<int> d = {1, 2, 3, 4, 5, 6};
    //从头部和尾部删除
    d.pop_front();//d.erase(d.begin());
    d.pop_back();//d.erase(d.end());
    //从中间删除
    d.erase(d.begin() + 1);
    for (deque<int>::iterator i = d.begin(); i != d.end(); ++i)
    {
        cout << (*i) << endl;
    }
    //清空所有元素
    d.clear();
    cout << d.size() << endl;
    return 0;
}

List

#include <iostream>
#include <list>
#include <algorithm>
#include <functional>

using namespace std;

int main()
{
    list<int> l = {2, 15, 20, 45};
    list<int> li = {1, 10, 20, 20, 20, 20, 30};
    cout << l.size() << endl;
    cout << l.front() << endl;
    cout << l.back() << endl;
    cout << "===============" << endl;
    li.unique();//去重复
    for (auto num : li)
    {
        cout << num << endl;
    }
    cout << "===============" << endl;
    //默认升序
    l.merge(li);
    for (list<int>::iterator i = l.begin(); i != l.end(); ++i)
    {
        cout << *i << endl;
    }
    cout << "===============" << endl;
    //降序排序必须保证,将两个list都是降序,合并之后li链表为空
    l.reverse();
    l.merge(li, std::greater<int>());
    for (list<int>::iterator i = l.begin(); i != l.end(); ++i)
    {
        cout << *i << endl;
    }
    cout << "===============" << endl;

    return 0;
}

splice方法

#include <iostream>
#include <list>

//操作符"<<"重载
std::ostream& operator<<(std::ostream& ostr, const std::list<int>& list)
{
    for (auto &i : list) {
        ostr << " " << i;
    }
    return ostr;
}
 
int main ()
{
    std::list<int> list1 = { 1, 2, 3, 4, 5 };
    std::list<int> list2 = { 10, 20, 30, 40, 50 };

    auto it = list1.begin();
    //迭代器向后移两位
    std::advance(it, 2);
    //将list2的所有元素放在(list1.begin() + 2)前
    list1.splice(it, list2);
 
    std::cout << "list1: " << list1 << "\n";
    std::cout << "list2: " << list2 << "\n";

    //从list1转移范围[it, list1.end())中的元素到list2中
    std::cout << *it << std::endl; // 3
    list2.splice(list2.begin(), list1, it, list1.end());
 
    std::cout << "list1: " << list1 << "\n";
    std::cout << "list2: " << list2 << "\n";
    return 0;
}
